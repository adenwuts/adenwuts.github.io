Forking

#include <unistd.h> /* execvp */
#include <stdio.h> /* perror */
#include <sys/types.h> /* pid_t */
#include <sys/wait.h> /* waitpid */

#define NUMARGS 80

int main (int argc, char *argv[]) {

    pid_t childpid;
    int status = 0;

    /*
      fork() returns the PID of the child process in the parent,
      and 0 in the child. On failure, -1 is returned in the parent.
    */
    
    if ((childpid = fork()) == -1) {
        perror("fork");
        exit(1);
    }

    /* Check if we're running in the child process */
    if (childpid == 0) {
        char **args = NULL;
        args = (char **)malloc(NUMARGS * sizeof(char *));
        args[0] = "/path/to/exec";
        execvp(args[0], args);
        perror(execvp);
    } else {
        /* else if we're running in the parent, wait for the child to close */
        waitpid(childpid, &status, 0);
    }

    return 0;
}
